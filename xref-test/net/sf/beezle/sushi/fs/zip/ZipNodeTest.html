42      @Test
43      public void junit() throws Exception {
44          FileNode jar;
45          String rootPath;
46          String locator;
47          ZipNode assrt;
48          ZipNode junit;
49          Node root;
50          List<? extends Node> list;
51          List<? extends Node> tree;
52  
53          jar = world.locateClasspathItem(Assert.class);
54          rootPath = jar.getURI().toString() + "!/org/junit/Assert.class";
55          locator = "jar:" + rootPath;
56          assrt = (ZipNode) world.node(locator);
57          assertEquals(locator, assrt.getURI().toString());
58          assertEquals("org/junit/Assert.class", assrt.getPath());
59          assertTrue(assrt.exists());
60          assertTrue(assrt.isFile());
61          assertTrue(assrt.readString().length() > 100);
62          junit = assrt.getParent();
63          assertEquals("org/junit", junit.getPath());
64          assertTrue(junit.isDirectory());
65          list = junit.list();
66          assertTrue(list.size() > 10);
67          assertTrue(list.contains(assrt));
68          assertFalse(list.contains(list));
69          assertEquals(2, junit.getParent().list().size());
70          root = junit.getParent().getParent();
71          assertEquals("", root.getPath());
72          assertTrue(root.isDirectory());
73          assertTrue(root.exists());
74          tree = junit.find("**/*");
75          assertTrue(tree.size() > list.size());
76          assertTrue(tree.contains(assrt));
77          assertFalse(tree.contains(list));
78          assertTrue(tree.containsAll(list));
79          assrt = junit.join("Assert.class");
80          assertTrue(assrt.exists());
81          assertTrue(assrt.isFile());
82      }

87          ZipNode root;
88          int count;

90          jar = world.locateClasspathItem(Assert.class);
91          root = jar.openZip();
92          count = 0;
93          for (Node node : root.find("**/*")) {
94              count++;
95              while (true) {
96                  node = node.getParent();
97                  if (node == null) {
98                      break;
99                  }
100                 assertTrue(node.isDirectory());

102             }
103         }

134     private void checkSpecialPath(String dir, String name) throws IOException {
135         final String clazz = "org/junit/Assert.class";
136         FileNode jar;
137         Node temp;
138         Node copy;
139         ZipNode zip;
140 
141         jar = world.locateClasspathItem(Assert.class);
142         temp = world.getTemp().createTempDirectory();
143         copy = temp.join(dir).mkdir().join(name);
144         jar.copyFile(copy);
145         zip = ((FileNode) copy).openZip();
146         assertEquals(1, zip.find(clazz).size());
147         assertNotNull(world.validNode("zip:" + copy.getURI() + "!/" + clazz).readBytes());
148         assertNotNull(world.validNode("jar:" + copy.getURI() + "!/" + clazz).readBytes());
149         temp.deleteTree();
150     }
