23  import static org.junit.Assert.assertEquals;

26  public class SelectorTest {
27      private static final Builder BUILDER = new Builder();
28      private static final Selector SELECTOR = new Selector();
29      
30      @Test
31      public void getIntNormal() throws XmlException {
32          assertEquals(42, getInt("42"));
33      }
34  
35      @Test
36      public void getIntNagative() throws XmlException {
37          assertEquals(-1, getInt("-1"));
38      }
39      @Test
40      public void getIntMax() throws XmlException {
41          assertEquals(Integer.MAX_VALUE, getInt("" + Integer.MAX_VALUE));
42      }
43      
44      @Test(expected=XmlException.class)
45      public void getOverflow() throws XmlException {
46          getInt("" + (((long) Integer.MAX_VALUE) + 1));
47      }
48  
49      @Test
50      public void getLong() throws XmlException {
51          Document doc;
52          long v;
53          
54          v = Integer.MAX_VALUE;
55          v++;
56          assertTrue(v > Integer.MAX_VALUE);
57          doc = BUILDER.literal("<root>" + v + "</root>");        
58          assertEquals(v, SELECTOR.longer(doc.getDocumentElement(), "."));
59      }
60  
61      private int getInt(String str) throws XmlException {
62          Document doc;
63          
64          doc = BUILDER.literal("<root>" + str + "</root>");        
65          return SELECTOR.integer(doc.getDocumentElement(), ".");
66      }
67

70      @Test 
71      public void string() throws XmlException {
72          Document doc;
73          
74          doc = BUILDER.literal(
75                  "<published time='1173806109'>" +
76                  "  <date format='RFC2822'>Tue, 13 Mar 2007 18:15:09 +0100</date>" +
77                  "  <date format='RFC3339'>2007-03-13T18:15:09+01:00</date>" +
78                  "</published>");
79          assertEquals("Tue, 13 Mar 2007 18:15:09 +0100", SELECTOR.string(doc, "published/date[@format='RFC2822']"));
80      }

82      @Test
83      public void simple() {
84          simple(false, "");
85          simple(false, "/");
86          simple(true, "a");
87          simple(true, "1");
88          simple(true, "x42");
89          simple(true, "a/");
90          simple(true, "a/b");
91          simple(false, "a//b");
92      }
