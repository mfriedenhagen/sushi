20  import java.io.IOException;
21  import java.io.StringWriter;
22  import java.io.Writer;
23  import java.util.List;

25  /**
26   * Parser and Serializer for csv file lines.  
27   * 
28   * Note:
29   * o whitespace is always significant because it's never stripped by OOCalc 
30   *   (except for digit-only values).
31   * o comments are not supported because OOCalc would consider them as strings.
32   * o a line has at least one cell
33   * 
34   * You have to implement some pre- or post-processing if you want comments of whitespace handling.
35   *
36   *
37   * Grammar
38   * 
39   * line    = cell (SEP cell)* newline ;
40   * cell    = values 
41   *         | QUOTE values QUOTE ;
42   * values  = value (VSEP value)* 
43   *         | EMPTY
44   *         | NULL ;
45   * value   = char* ;
46   * char    = QUOTE QUOTE
47   *         | ESC CH
48   *         | SEP                               <-- aber nur wenn drumrum quotes stehen
49   *         | CH \ QUOTE \ ESC \ SEP \ VSEP  ;
50   * newline = CR | LF | CR LF ;
51   * 
52   * Terminals as configured for the format: 
53   *   QUOTE, SEP, VSEP, NEWLINE, COMMENT, NULL, EMPTY, ESC
54   *   
55   * Other Terminals:
56   *   CH: arbitrary unicode character
57   */

102     public Line read(String line) throws CsvLineException {
103         Line result;
104         Source src;
105         int c;
106         
107         result = new Line();
108         src = new Source(line);
109         while (true) {
110             cell(src, result);
111             c = src.peek();
112             if (c == Source.END) {
113                 return result;
114             }
115             if (c != separator) {
116                 throw new CsvLineException("separator expected");
117             }
118             src.eat();
119         }
120     }
121 
122     private void cell(Source src, Line result) throws CsvLineException {
123         boolean quoted;
124         
125         quoted = src.peek() == quote;
126         if (quoted) {
127             src.eat();
128         }
129         values(src, quoted, result);
130         if (quoted) {
131             if (src.peek() != quote) {
132                 throw new CsvLineException("quote not closed");
133             }
134             src.eat();
135         }
136     }
137 
138     private void values(Source src, boolean quoted, Line result) throws CsvLineException {
139         List<String> values;
140 
141         if (src.eat(nul, quoted ? quote : separator)) {
142             result.addNull();
143             return;
144         }
145         values = result.add();
146         if (src.eat(empty, quoted ? quote : separator)) {
147             return;
148         }
149         while (true) {
150             value(src, quoted, values);
151             if (src.peek() != valueSeparator) {
152                 return;
153             }
154             src.eat();
155         }
156     }
157 
158     private void value(Source src, boolean quoted, List<String> values) throws CsvLineException { 
159         StringBuilder builder;
160         int c;
161         
162         builder = new StringBuilder();
163         while (true) {
164             c = src.peek();
165             if (c == -1) {
166                 break;
167             } else if (c == quote) {
168                 if (src.peekNext() == quote) {
169                     src.eat();
170                     builder.append(quote);
171                 } else {
172                     break;
173                 }
174             } else if (c == separator) {
175                 if (quoted) {
176                     builder.append(c);
177                 } else {
178                     break;
179                 }
180             } else if (c == valueSeparator) {
181                 break;
182             } else if (c == escape) {
183                 src.eat();
184                 c = src.peek();
185                 if (c == Source.END) {
186                     throw new CsvLineException("tailing escape character " + escape);
187                 }
188                 builder.append((char) c);
189             } else {
190                 builder.append((char) c);
191             }
192             src.eat();
193         }
194         values.add(builder.toString());
195     }

199     public String write(Line line) {
200         StringWriter dest;
201         
202         dest = new StringWriter();
203         try {
204             write(line, dest);
205         } catch (IOException e) {
206             throw new RuntimeException("unused", e);
207         }
208         return dest.toString();
209     }

211     public void write(Line line, Writer dest) throws IOException {
212         for (int i = 0, max = line.size(); i < max; i++) {
213             if (i > 0) {
214                 dest.write(separator);
215             }
216             writeCell(line.get(i), dest);
217         }
218         dest.write('\n');
219     }
220     
221     private void writeCell(List<String> values, Writer dest) throws IOException {
222         boolean quoted;
223         char c;
224         boolean first;
225 
226         if (values == null) {
227             dest.write(nul);
228             return;
229         }
230         if (values.size() == 0) {
231             dest.write(empty);
232             return;
233         }
234         quoted = values.size() != 1 || needsQuotes(values.get(0));
235         if (quoted) {
236             dest.write(quote);
237         }
238         first = true;
239         for (String value : values) {
240             if (first) {
241                 first = false;
242             } else {
243                 dest.write(valueSeparator);
244             }
245             if (value.equals(nul)) {
246                 dest.write(escape);
247                 dest.write(nul);
248             } else if (value.equals(empty)) {
249                 dest.write(escape);
250                 dest.write(empty);
251             } else {
252                 for (int i = 0, max = value.length(); i < max; i++) {
253                     c = value.charAt(i);
254                     if (c == quote) { 
255                         dest.write(quote);
256                         dest.write(quote);
257                     } else if (c == separator || c == valueSeparator || c == escape) {
258                         dest.write(escape);
259                         dest.write(c);
260                     } else {
261                         dest.write(c);
262                     }
263                 }
264             }
265         }
266         if (quoted) {
267             dest.write(quote);
268         }
269     }
270 
271     private boolean needsQuotes(String value) {
272         for (int i = 0, max = value.length(); i < max; i++) {
273             if (unquoted.indexOf(value.charAt(i)) == -1) {
274                 return true;
275             }
276         }
277         return false;
278     }
