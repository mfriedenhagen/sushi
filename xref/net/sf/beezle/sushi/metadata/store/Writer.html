18  package net.sf.beezle.sushi.metadata.store;
19  
20  import net.sf.beezle.sushi.metadata.Cardinality;
21  import net.sf.beezle.sushi.metadata.ComplexType;
22  import net.sf.beezle.sushi.metadata.Item;
23  import net.sf.beezle.sushi.metadata.SimpleType;
24  import net.sf.beezle.sushi.metadata.Type;
25  
26  import java.util.ArrayList;
27  import java.util.Collection;
28  import java.util.List;
29  
30  /**
31   * Helper class to read and write properties. You'll usually not use this class directly,
32   * use Type.loadProperties and Data.toProperties instead. 
33   */
34  public class Writer {
35      public static void write(Type type, Object obj, String name, final Store dest) {
36          new Writer(dest).write(new ArrayList<Item<?>>(), type, obj, name);
37      }
38  
39      private final Store dest;
40      
41      public Writer(Store dest) {
42          this.dest = dest;
43      }
44  
45      public Type write(List<Item<?>> parents, Type type, Object obj, String path) {
46          if (obj != null) {
47              type = type.getSchema().type(obj.getClass());
48          }
49          writeThis(parents, type, obj, path);
50          if (type instanceof ComplexType) {
51              ComplexType complex;
52              String childName;
53              Collection<?> children;
54              int idx;
55  
56              complex = (ComplexType) type;
57              for (Item item : complex.items()) {
58                  children = item.get(obj);
59                  parents.add(item);
60                  idx = 0;
61                  for (Object child : children) {
62                      childName = join(path, item.getName());
63                      if (item.getCardinality() == Cardinality.SEQUENCE) {
64                          childName = childName + "[" + Integer.toString(idx++) + "]";
65                      }
66                      write(parents, item.getType(), child, childName);
67                  }
68                  parents.remove(parents.size() - 1);
69              }
70          }
71          return type;
72      }
73  
74      protected void writeThis(List<Item<?>> parents, Type type, Object obj, String path) {
75          String value;
76          
77          if (type instanceof SimpleType) {
78              value = ((SimpleType) type).valueToString(obj);
79          } else if (obj == null) {
80              value = type.getType().getName();
81          } else {
82              value = obj.getClass().getName(); 
83          }
84          try {
85              dest.write(parents, path, value);
86          } catch (RuntimeException e) {
87              throw e;
88          } catch (Exception e) {
89              throw new StoreException(path + ": write failed: " + e.getMessage(), e);
90          }
91      }
92  
93      private String join(String first, String second) {
94          return first.length() == 0 ? second : first + "/" + second;
95      }
96  }
